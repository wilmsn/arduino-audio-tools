131  bool begin(A2DPConfig cfg){
132  this->config = cfg;
133  bool result = false;
134  LOGI("Connecting to %s",cfg.name);
135  a2dp_buffer.resize(cfg.buffer_size);
136 
137  // initialize a2dp_silence_timeout

139  LOGI("Using StartOnConnect")
140  config.startup_logic = StartOnConnect;

143  switch (cfg.mode){
144  case TX_MODE:
145  LOGI("Starting a2dp_source...");
146  source(); // allocate object
147  a2dp_source->set_auto_reconnect(cfg.auto_reconnect);
148  a2dp_source->set_volume(volume() * A2DP_MAX_VOL);

150  //search next available device

154  a2dp_source->start_raw((char*)cfg.name, a2dp_stream_source_sound_data);
155  while(!a2dp_source->is_connected()){
156  LOGD("waiting for connection");
157  delay(1000);
158  }
159  LOGI("a2dp_source is connected...");
160  notify_base_Info(44100);
161  //is_a2dp_active = true;
162  result = true;
163  break;
164 
165  case RX_MODE:
166  LOGI("Starting a2dp_sink...");
167  sink(); // allocate object
168  a2dp_sink->set_auto_reconnect(cfg.auto_reconnect);
169  a2dp_sink->set_stream_reader(&a2dp_stream_sink_sound_data, false);
170  a2dp_sink->set_volume(volume() * A2DP_MAX_VOL);

172  a2dp_sink->set_sample_rate_callback(sample_rate_callback);

175  LOGD("waiting for connection");

215  size_t write(const uint8_t* data, size_t len) override {
216  LOGD("%s: %zu", LOG_METHOD, len);
217  if (config.mode == TX_MODE){
218  // at 80% we activate the processing
219  if(!is_a2dp_active
220  && config.startup_logic == StartWhenBufferFull
221  && a2dp_buffer.available() >= 0.8f * a2dp_buffer.size()){
222  LOGI("set active");
223  is_a2dp_active = true;
224  }
225 
226  // blocking write: if buffer is full we wait
227  while(len > 0.9f * a2dp_buffer.availableForWrite() * 100 / 90){
228  LOGD("Waiting for buffer to be available");
229  delay(5);
230  }
231  }
232 
233  // write to buffer
234  size_t result = a2dp_buffer.writeArray(data, len);
235  LOGD("write %d -> %d", len, result);
236  if (config.mode == TX_MODE){

300  // callback used by A2DP to provide the a2dp_source sound data
301  static int32_t a2dp_stream_source_sound_data(uint8_t* data, int32_t len) {
302  int32_t result_len = 0;
303  A2DPConfig config = A2DPStream_self->config;
304 
305  // at first call we start with some empty data
306  if (is_a2dp_active){
307  // the data in the file must be in int16 with 2 channels
308  yield();
309  result_len = a2dp_buffer.readArray((uint8_t*)data, len);
310 
311  // provide silence data
312  if (config.silence_on_nodata && result_len == 0){
313  memset(data,0, len);
314  result_len = len;
315  }
316  } else {
317 
318  // prevent underflow on first call
319  switch (config.startup_nodata) {
320  case A2DPSilence:
321  memset(data, 0, len);
322  break;
323  case A2DPWhoosh:
324  int16_t *data16 = (int16_t*)data;
325  for (int j=0;j<len/4;j+=2){
326  data16[j+1] = data16[j] = (rand() % 50) - 25;
327  }
328  break;
329  }
330  result_len = len;
331 
332  // Priority: 22 on core 0
333  // LOGI("Priority: %d on core %d", uxTaskPriorityGet(NULL), xPortGetCoreID());
334 
335  }
336  LOGD("a2dp_stream_source_sound_data: %d -> %d", len, result_len);
337  return result_len;
338  }
