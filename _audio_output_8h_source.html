408  void flushMixer() {
409  LOGD("flush");
410  bool result = false;
411 
412  // determine ringbuffer with mininum available data
413  size_t samples = size_bytes / sizeof(T);
414  for (int j = 0; j < output_count; j++) {
415  samples = MIN(samples, (size_t)buffers[j]->available());
416  }
417 
418  if (samples > 0) {
419  result = true;
420  // mix data from ringbuffers to output
421  output.resize(samples);
422  memset(output.data(), 0, samples * sizeof(T));
423  for (int j = 0; j < output_count; j++) {
424  float weight = weights[j];
425  // sum up input samples to result samples
426  for (int i = 0; i < samples; i++) {
427  output[i] += weight * buffers[j]->read() / total_weights;
428  }
429  }
430 
431  // write output
432  LOGD("write to final out: %d", samples * sizeof(T));
433  p_final_output->write((uint8_t *)output.data(), samples * sizeof(T));
434  }
435  stream_idx = 0;
436  return;
437  }

712  ChannelSelectionOutputDef &def = out_channels[ch];

716  result[i++] = data[j];

718  // write mono result

720  def.p_out->write((uint8_t *)result, result_size * sizeof(T));
