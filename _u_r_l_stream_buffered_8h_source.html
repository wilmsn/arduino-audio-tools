119  void processTask() {
120  size_t available_to_write = this->buffers.availableForWrite();
121  if (*(this->p_stream) && available_to_write > 0) {
122  size_t to_read = min(available_to_write, (size_t)512);
123  uint8_t buffer[to_read];
124  size_t avail_read = this->p_stream->readBytes((uint8_t *)buffer, to_read);
125  size_t written = this->buffers.writeArray(buffer, avail_read);
126 
127  if (written != avail_read) {
128  LOGE("DATA Lost! %zu reqested, %zu written!", avail_read, written);
129  }
130 
131  } else {
132  // 3ms at 44100 stereo is about 529.2 bytes
133  delay(3);
134  }
135  // buffer is full we start to provide data
136  if (available_to_write == 0) {
137  this->ready = true;
138  }
