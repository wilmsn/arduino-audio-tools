137  TRACED();

142  available_buffers.resize(bufferCount);

144 
145  setReadMaxWait(readMaxWait);

147

149  NBuffer<T>::write_buffer_count = 0;
150  for (int j = 0; j < bufferCount; j++) {
151  BaseBuffer<T> *tmp = new SingleBuffer<T>(bufferSize);
152  if (tmp != nullptr) {
